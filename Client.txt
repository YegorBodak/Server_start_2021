#pragma comment(lib, "ws2_32.lib")
#include <winsock2.h>
#include <iostream>
#include <string>
#pragma warning(disable: 4996)
#include <Windows.h>
#include <vector>

using namespace std;

SOCKET Connection;

void ClientHandler() {
	int msg_size;
	while (true) {
		recv(Connection, (char*)&msg_size, sizeof(int), NULL);
		char* msg = new char[msg_size + 1];
		msg[msg_size] = '\0';
		recv(Connection, msg, msg_size, NULL);
		cout << "||chat client: " << msg << std::endl;
		delete[] msg; 
	}
}

int main(int argc, char* argv[]) {
	system("color E");
	
	string ip; // ip servera

	cout << "[----------Start Client----------]\n" << endl;
	cout << " ~Enter ip servera: ";
	cin >> ip;

	//WSAStartup
	WSAData wsaData;
	WORD DLLVersion = MAKEWORD(2, 1);
	if (WSAStartup(DLLVersion, &wsaData) != 0) {
		cout << "Error not version compatibility!" << endl;
		Sleep(10000);
		//exit(1);
	}

	SOCKADDR_IN addr;
	int sizeofaddr = sizeof(addr);
	addr.sin_addr.s_addr = inet_addr(ip.data()); // 10.10.201.56
	addr.sin_port = htons(333);
	addr.sin_family = AF_INET;

	Connection = socket(AF_INET, SOCK_STREAM, NULL);
	

	while (connect(Connection, (SOCKADDR*)&addr, sizeof(addr)) != 0) {
		
		cout << "\n\nError: failed connect to server.\n";
		cout << " ~Please enter a new ip: ";
		cin >> ip;
		Sleep(100);
		addr.sin_addr.s_addr = inet_addr(ip.data());
	}
	cout << "Connected!\n";

	CreateThread(NULL, NULL, (LPTHREAD_START_ROUTINE)ClientHandler, NULL, NULL, NULL);

	

	std::string msg1;
	while (true) {
		cout << "MSG : ";
		getline(std::cin, msg1);
		int msg_size = msg1.size();
		send(Connection, (char*)&msg_size, sizeof(int), NULL);
		send(Connection, msg1.c_str(), msg_size, NULL);
		Sleep(10);
	}

	//system("pause");
	Sleep(10000);
	return 0;
}
